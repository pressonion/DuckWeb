let loadTimer; function playGame(gameUrl) { clearTimeout(loadTimer); openMenu('welcome-screen'); // Hide other menus document.getElementById('welcome-screen').style.display = 'none'; const sidebar = document.getElementById('sidebar'); sidebar.style.display = 'none'; setTimeout(() => { sidebar.style.display = ''; }, 100); const loader = document.getElementById('loader'); loader.style.display = 'flex'; loadTimer = setTimeout(() => { loader.style.display = 'none'; const frame = document.getElementById('game-frame'); frame.style.display = 'block'; // --- THE CACHE BUSTER FIX --- // This adds a timestamp to the URL so the browser thinks it's a brand new page every time! const cacheBuster = "?v=" + new Date().getTime(); frame.src = gameUrl + cacheBuster; frame.focus(); // Give iframe focus for the AFK tracker }, 2000); }
